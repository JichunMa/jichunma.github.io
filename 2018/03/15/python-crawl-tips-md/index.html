multiprocessing.Pool 传递多个参数

def low_work(t, s):
    print("%d %d" % (t, s))
    time.sleep(t)

if __name__ == '__main__':
    pool = Pool(4)
    tasks = [(x, y) for x in range(4,8) for y in range(10, 14)]
    pool.starmap(low_work, tasks)
    pool.close()
